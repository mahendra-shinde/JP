Q1. Write SQL Query to create following tables Customer and Account

Create Table Customer
(
    CustomerId varchar2(10), 
    Fullname VARCHAR2(40),
    Address Varchar2(80), 
    city varchar2(20), 
    PAN_Number varchar2(20)
);

Create Table Account
(
    accountNo varchar2(20), 
    accType VARCHAR2(10),
    balance Number, 
    customerID varchar2(10)
);
